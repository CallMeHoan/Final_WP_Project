create   proc [dbo].[SP_UpdateBooking]
@empID nchar(10), @RoomID nchar(10), @CusID nchar(10), @start time, @end time, @date date
as
begin
	Update Booking set State = 'Paid' 
	where
		EmployeeID = @empID and 
		RoomID = @RoomID and 
		CustomerID = @CusID and 
		StartTime = @start and 
		EndTime = @end and 
		Date = @date
end
GO
